(a) ELU¶

Output range

  • when x ≤ 0, ELU(x) = 0.1(exp(x) - 1), since exp(x) ∈ (0, 1], output range is (-0.1, 0]
  • when x > 0, ELU(x) = x, output range is (0, +∞)
  • Therefore, the whole output range of ELU is (-0.1, +∞)

Derivative

  • when x ≤ 0, $\frac{d}{dx}[ELU(x)]$ = $\frac{d}{dx}[0.1(exp(x) - 1)]$ = 0.1 * exp(x)
  • when x > 0, $\frac{d}{dx}[ELU(x)]$ = $\frac{d}{dx}[x]$ = 1

Final derivative

$$\mathrm{ELU}'(x) =\begin{cases} 0.1\exp(x) & \text{if } x \leq 0 \\ 1 & \text{if } x > 0 \end{cases}$$

(b) GELU¶

Output range

  • Since Φ(x) ∈ [0, 1] and x ∈ (-∞, +∞)
  • when x → -∞, GELU(x) → 0
  • when x → +∞, GELU(x) → x
  • Hence, the output range of GELU is (-∞, +∞)

Derivative

According to the Product rule, we have $$ \frac{d}{dx}[GELU(x)] = \mathrm{(x)}'·Φ(x) + \mathrm{Φ}'(x)·x $$ and we have: $$ {Φ(x)}=\tfrac{1}{\sqrt{2\pi}}e^{-x^2/2}\ $$

Final derivative

$$\mathrm{GELU}'(x) = \Phi(x) + x \cdot \phi(x)$$

(g)¶

Definition of CE：

$$ \mathrm{CE}(y,p(x)) = -\sum_{i=1}^{M} y_i \log p_i(x) $$

since $y$ is one-hot vector, if the ground truth is $k$, we have：

$$ \mathrm{CE}(y,p(x)) = -\log p_k(x) $$

Reasone of Non-negative：

$$ 0 \le p_k(x) \le 1 \;\;\Longrightarrow\;\; \log p_k(x) \le 0 \;\;\Longrightarrow\;\; -\log p_k(x) \ge 0 $$

Thus：

$$ \mathrm{CE}(y,p(x)) \ge 0 $$

When equal to 0：

If and only if:

$$ p_k(x) = 1 $$

in this case：

$$ \mathrm{CE}(y,p(x)) = -\log 1 = 0 $$

class MyFFN(torch.nn.Module):
    def __init__(self, input_size, num_classes, hidden_sizes, act=torch.nn.ReLU()):
        """
        input_size: the size of the input
        num_classes: the number of classes
        act is the activation function
        hidden_sizes is the list of hidden sizes
        for example input_size = 3, hidden_sizes = [5, 7], num_classes = 4, and act = torch.nn.ReLU()
        means that we are building up a FFN with the confirguration
        (3 (Input) -> 5 (ReLU) -> 7 (ReLU) -> 4 (Output))
        """
        super(MyFFN, self).__init__()
        self.input_size = input_size
        self.num_classes = num_classes
        self.act = act
        self.hidden_sizes = hidden_sizes
        self.num_layers = len(hidden_sizes) + 1
        self._eps = 1e-8  # avoid division by zero
        self._vel = {}
        self.square_grad = {}

    def create_FFN(self):
        """
        This function creates the feed-forward neural network
        We stack many MyLinear layers
        """
        hidden_sizes = [self.input_size] + self.hidden_sizes + [self.num_classes]
        self.layers = []
        # Your code here
        for i in range(len(hidden_sizes) - 1):
            self.layers.append(MyLinear(hidden_sizes[i], hidden_sizes[i + 1]))
        self.layers = nn.ModuleList(self.layers)

    def forward(self, x):
        """
        This implements the forward propagation of the batch x
        This needs to return the logits of x
        """
        # Your code here
        # first n-1 layers: Linear + ReLU
        for i, layer in enumerate(self.layers):
            x = layer(x)
            if i < len(self.layers) - 1:
                self.act(x)
        return x

    def compute_loss(self, x, y):
        """
        This function computes the cross-entropy loss
        You can use the built-in CE loss of PyTorch
        """
        # Your code here
        logits = self.forward(x)
        return nn.functional.cross_entropy(logits, y)

    def update_SGD(self, x, y, learning_rate=0.01):
        """
        This function updates the model parameters using SGD using the batch (x,y)
        You need to implement the update rule manually and cannot rely on the built-in optimizer
        """
        # Your code here
        # forward and loss
        loss = self.compute_loss(x, y)
        # clear grad
        self.zero_grad(set_to_none=True)
        # backward
        loss.backward()

        # theta <- theta - lr * grad
        with torch.no_grad():
            for p in self.parameters():
                if p.grad is None:
                    continue
                p -= learning_rate * p.grad

        # return loss.item()

    def update_SGDwithMomentum(self, x, y, learning_rate=0.01, momentum=0.9):
        """
        This function updates the model parameters using SGD with momentum using the batch (x,y)
        You need to implement the update rule manually and cannot rely on the built-in optimizer
        """
        # Your code here
        loss = self.compute_loss(x, y)
        self.zero_grad(set_to_none=True)
        loss.backward()

        with torch.no_grad():
            for p in self.parameters():
                if p.grad is None:
                    continue
                # velocity initialization
                if p not in self._vel:
                    self._vel[p] = torch.zeros_like(p)
                # velocity update
                self._vel[p] = momentum * self._vel[p] + p.grad
                p -= learning_rate * self._vel[p]

    def update_AdaGrad(self, x, y, learning_rate=0.01):
        """
        This function updates the model parameters using AdaGrad using the batch (x,y)
        You need to implement the update rule manually and cannot rely on the built-in optimizer
        """
        # Your code here
        loss = self.compute_loss(x, y)
        self.zero_grad(set_to_none=True)
        loss.backward()

        with torch.no_grad():
            for name, p in self.named_parameters():
                if p.grad is None:
                    continue
                if name not in self.square_grad:
                    self.square_grad[name] = torch.zeros_like(p)
                self.square_grad[name] += p.grad * p.grad
                denom = torch.sqrt(self.square_grad[name]) + self._eps
                p -= learning_rate * p.grad / denom

def compute_acc(model, data_loader):
    """
    This function computes the accuracy of the model on a data loader
    """
    # Your code here
    model.eval()
    correct = 0
    total = 0
    with torch.no_grad():
        for x, y in data_loader:
            x = x.view(x.size(0), -1).float()
            y = y.long()
            outputs = model(x)
            _, pred = torch.max(outputs, 1)
            total += y.size(0)  # number of samples
            correct += (pred == y).sum().item()  # number of correct predictions
    model.train()

    return correct / total if total > 0 else 0.0

def compute_loss(model, data_loader):
    """
    This function computes the loss of the model on a data loader
    """
    # Your code here
    model.eval()
    total_loss = 0.0
    total_samples = 0
    with torch.no_grad():
        for x, y in data_loader:
            x = x.view(x.size(0), -1).float()
            y = y.long()
            loss = model.compute_loss(x, y)
            batch_size = y.size(0)
            total_loss += loss.item() * batch_size
            total_samples += batch_size

    model.train()

    return total_loss / total_samples if total_samples > 0 else 0.0
